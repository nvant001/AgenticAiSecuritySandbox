C++ Development workflow
Create a new folder and open it in vscode
Open terminal and run commands:
    vcpkg new --name [project-name-here] --version 0.1.0
**Add dependencies**
    vcpkg add port [dependency-name "eigen3"]

Create CMakeLists.txt
//
cmake_minimum_required(VERSION 3.15)
project(MyProjectName)

# Enable M_PI and other math constants for MSVC
add_compile_definitions(_USE_MATH_DEFINES)

# Use C++20 for modern features (like std::numbers::pi)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find and Link Libraries (vcpkg bridge handles this)
find_package(Eigen3 CONFIG REQUIRED)

add_executable(${PROJECT_NAME} main.cpp)
target_link_libraries(${PROJECT_NAME} PRIVATE Eigen3::Eigen)
//
